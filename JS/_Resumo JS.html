Altere o texto do cabeçalho para "Tabela de preços" let tituloPag = document.querySelector('#cabec'); // Opção: document.getElementById('cabec') tituloPag.innerText = 'Tabela de preços'; //2. Coloque no primeiro parágrafo a quantia total necessária // (calculada em código) para comprar 1kg de cada fruta let precos = document.querySelectorAll('.alin-dir'); // Alternativa: // let precos = document.getElementsByClassName('alin-dir'); let total = 0; for(let preco of precos) { total += parseFloat(preco.innerText); } // Obtendo os dois parágrafos de uma única vez em um único vetor let parags = document.getElementsByTagName('p'); // Alternativa: // let parags = document.querySelectorAll('p'); // 1º parágrafo (índice 0) parags[0].innerText = `Valor total requirido: ${total}`; // 3.
